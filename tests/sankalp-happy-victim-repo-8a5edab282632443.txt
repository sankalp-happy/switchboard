const express = require("express");
const path = require("path");
const fs = require("fs");
const sqlite3 = require("sqlite3").verbose();

const app = express();
const PORT = process.env.PORT || 3000;

const dataDir = path.join(__dirname, "data");
const dbPath = path.join(dataDir, "lab.db");

if (!fs.existsSync(dataDir)) {
  fs.mkdirSync(dataDir, { recursive: true });
}

const db = new sqlite3.Database(dbPath);

function seedDatabase() {
  db.serialize(() => {
    db.run(
      "CREATE TABLE IF NOT EXISTS users (id INTEGER PRIMARY KEY AUTOINCREMENT, username TEXT, password TEXT, email TEXT, ssn TEXT, bio TEXT)"
    );
    db.run(
      "CREATE TABLE IF NOT EXISTS comments (id INTEGER PRIMARY KEY AUTOINCREMENT, author TEXT, message TEXT, created_at TEXT)"
    );

    db.get("SELECT COUNT(*) as count FROM users", (err, row) => {
      if (err) {
        return;
      }

      if (row.count === 0) {
        const seedUsers = [
          ["alice", "password123", "alice@acme.local", "100-11-2222", "Product owner"],
          ["bob", "letmein", "bob@acme.local", "200-22-3333", "Finance manager"],
          ["charlie", "qwerty", "charlie@acme.local", "300-33-4444", "Support lead"]
        ];

        const stmt = db.prepare("INSERT INTO users (username, password, email, ssn, bio) VALUES (?, ?, ?, ?, ?)");
        for (const user of seedUsers) {
          stmt.run(user);
        }
        stmt.finalize();
      }
    });
  });
}

seedDatabase();

app.post("/api/login", (req, res) => {
  const username = req.body.username || "";
  const password = req.body.password || "";

  const query = `SELECT id, username, email FROM users WHERE username = '${username}' AND password = '${password}' LIMIT 1`;

  db.get(query, (err, row) => {
    if (err) {
      return res.status(500).json({ error: err.message, query });
    }

    if (!row) {
      return res.status(401).json({ success: false, message: "Invalid credentials", query });
    }

    res.json({
      success: true,
      message: "Login successful",
      user: row,
      query
    });
  });
});

app.get("/api/account/:id", (req, res) => {
  const accountId = req.params.id;
  const query = `SELECT id, username, email, ssn, bio FROM users WHERE id = ${accountId}`;

  db.get(query, (err, row) => {
    if (err) {
      return res.status(500).json({ error: err.message, query });
    }

    if (!row) {
      return res.status(404).json({ message: "Account not found", query });
    }

    res.json({
      account: row,
      note: "No ownership validation is performed on this object.",
      query
    });
  });
});

app.post("/api/comments", (req, res) => {
  const author = req.body.author || "anonymous";
  const message = req.body.message || "";
  const createdAt = new Date().toISOString();

  db.run(
    "INSERT INTO comments (author, message, created_at) VALUES (?, ?, ?)",
    [author, message, createdAt],
    function insertCallback(err) {
      if (err) {
        return res.status(500).json({ error: err.message });
      }

      res.status(201).json({
        id: this.lastID,
        author,
        message,
        created_at: createdAt
      });
    }
  );
});

app.get("/api/comments", (_, res) => {
  db.all("SELECT id, author, message, created_at FROM comments ORDER BY id DESC LIMIT 20", (err, rows) => {
    if (err) {
      return res.status(500).json({ error: err.message });
    }

    res.json(rows);
  });
});

app.get("/api/search", (req, res) => {
  const term = req.query.term || "";

app.get("/redirect", (req, res) => {

const loginForm = document.getElementById("login-form");

const loginResult = document.getElementById("login-result");

const commentsList = document.getElementById("comments");
const searchResult = document.getElementById("search-result");
const corsResult = document.getElementById("cors-result");

function printJson(node, data) {
  node.textContent = JSON.stringify(data, null, 2);
}

loginForm.addEventListener("submit", async (event) => {
  event.preventDefault();
  const form = new FormData(loginForm);

  const response = await fetch("/api/login", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      username: form.get("username"),
      password: form.get("password")
    })
  });

  const data = await response.json();
  printJson(loginResult, data);
});

idorForm.addEventListener("submit", async (event) => {
  event.preventDefault();
  const form = new FormData(idorForm);
  const accountId = form.get("accountId");

  const response = await fetch(`/api/account/${accountId}`);
  const data = await response.json();
  printJson(idorResult, data);
});

commentForm.addEventListener("submit", async (event) => {
  event.preventDefault();
  const form = new FormData(commentForm);

  await fetch("/api/comments", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      author: form.get("author"),
      message: form.get("message")
    })
  });

  commentForm.reset();
  loadComments();
});

async function loadComments() {
  const response = await fetch("/api/comments");
  const comments = await response.json();

  commentsList.innerHTML = comments
    .map(
      (comment) => `
      <div class="comment">
        <strong>${comment.author}</strong>
        <p>${comment.message}</p>
        <small>${comment.created_at}</small>
      </div>
    `
    )
    .join("");
}

searchForm.addEventListener("submit", async (event) => {
  event.preventDefault();
  const form = new FormData(searchForm);
  const term = form.get("term");

  const response = await fetch(`/api/search?term=${encodeURIComponent(term)}`);
  const html = await response.text();

  searchResult.innerHTML = html;
});

redirectForm.addEventListener("submit", (event) => {
  event.preventDefault();
  const form = new FormData(redirectForm);
  const next = form.get("next");
  window.location.href = `/redirect?next=${encodeURIComponent(next)}`;
});

corsButton.addEventListener("click", async () => {
  const response = await fetch("/api/account/1", {
    credentials: "include"
  });

  const data = await response.json();

  printJson(corsResult, {
    note: "If requested from another origin, server reflects Origin and allows credentials.",
    sampleResponse: data
  });
});

loadComments();
